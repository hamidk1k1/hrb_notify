اچ آر بی نوتیفای

نوتیفیکیشن پیشرفته جهت نمایش پیغام سفارشی

نحوه کار کتابخانه (HRB_NOTIFY)

            hrb_notify([
                'نوع پیغام',
                "متن پیغام",
                'موقعیت نمایش پیغام',
                'افکت ظاهر شدن پیغام',
                'افکت حذف شدن پیغام',
                'تایمر حذف پیغام'
            ]);
  1. نوع پیغام می تونه مقادیر زیر رو بگیره
    • danger
    • warning
    • info
    • success
  2. متن پیغام متن مورد نظر شما هستش
  3. موقعیت نمایش پیغام مقادیر زیر رو میگیره

    • topCenter
    • topLeft
    • middleLeft
    • middleCenter
    • middleRight
    • bottomLeft
    • bottomCenter
    • bottomRight

نوع خطا

            hrb_notify([
                'danger',
                "نام کاربری وارد شده اشتباه است!",
                'topRight',
                'bounceIn',
                'bounceOut',
                '5'
            ]);
        
            hrb_notify([
                'warning',
                "سشن سایت منقضی شده",
                'topLeft',
                'bounceIn',
                'bounceOut',
                '5'
            ]);
        
            hrb_notify([
                'info',
                "کاربر گرامی ، حمیدرضابیگلری عزیز خوش آمدی",
                'bottomLeft',
                'rollIn',
                'rollOut',
                '5'
            ]);
        
            hrb_notify([
                'success',
                "با موفقیت وارد شدید",
                'bottomRight',
                'flipInY',
                'flipOutX',
                '5'
            ]);
